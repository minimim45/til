class Main {
  public static void main(String[] args) {
    int number = 11;
    String text = "Ruby";
    System.out.println(number);
    System.out.println(text);
    
    // 変数numberを9で上書きしてください
    number = 9; //変数をつけない（int）
    
    // 変数numberを出力してください
    System.out.println(number);
    
    // 変数textを「Java」で上書きしてください
    text = "Java";//書き換えるときは変数をつけない（String）
    
    // 変数textを出力してください
    System.out.println(text);
    
  }
}
